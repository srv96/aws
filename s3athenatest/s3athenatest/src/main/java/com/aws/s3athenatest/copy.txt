import java.util.List;

public class PreparedStatementInput {
    private final List<PreparedStatementParameter> parameters;

    public PreparedStatementInput(List<PreparedStatementParameter> parameters) {
        this.parameters = parameters;
    }

    public List<PreparedStatementParameter> getParameters() {
        return parameters;
    }
}

public class PreparedStatementParameter {
    private final int index;
    private final Object value;

    public PreparedStatementParameter(int index, Object value) {
        this.index = index;
        this.value = value;
    }

    public int getIndex() {
        return index;
    }

    public Object getValue() {
        return value;
    }
}
