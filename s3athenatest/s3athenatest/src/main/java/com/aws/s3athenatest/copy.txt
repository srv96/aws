import com.datadoghq.metrics.DogStatsd;

public class Main {
    public static void main(String[] args) {
        // Replace these values with your actual Datadog API and APP keys
        String apiKey = "YOUR_API_KEY";
        String appKey = "YOUR_APP_KEY";

        // Initialize DogStatsd client
        DogStatsd statsd = new DogStatsd.Builder()
                .apiKey(apiKey)
                .appKey(appKey)
                .build();

        // Send a custom log message
        statsd.event("Test log message", "This is a test log message", "success");

        // Close the client
        statsd.close();
    }
}
