import java.util.ArrayList;
import java.util.List;

public class PreparedStatementInput {
    private final List<PreparedStatementParameter> parameters;

    private PreparedStatementInput(Builder builder) {
        this.parameters = builder.parameters;
    }

    public List<PreparedStatementParameter> getParameters() {
        return parameters;
    }

    public static Builder builder() {
        return new Builder();
    }

    public static class Builder {
        private final List<PreparedStatementParameter> parameters = new ArrayList<>();

        public Builder setString(int index, String value) {
            parameters.add(new PreparedStatementParameter(index, value));
            return this;
        }

        public Builder setInt(int index, int value) {
            parameters.add(new PreparedStatementParameter(index, value));
            return this;
        }

        public PreparedStatementInput build() {
            return new PreparedStatementInput(this);
        }
    }
}
