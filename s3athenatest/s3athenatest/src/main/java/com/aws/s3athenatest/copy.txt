import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;

public class DummyDataGenerator {

    public static Map<String, HashSet<String>> createDummyMap() {
        Map<String, HashSet<String>> dummyMap = new HashMap<>();

        // Adding some dummy data to the map
        HashSet<String> set1 = new HashSet<>();
        set1.add("Value1A");
        set1.add("Value1B");
        set1.add("Value1C");

        HashSet<String> set2 = new HashSet<>();
        set2.add("Value2A");
        set2.add("Value2B");

        dummyMap.put("Key1", set1);
        dummyMap.put("Key2", set2);

        return dummyMap;
    }

    public static void main(String[] args) {
        // Example usage
        Map<String, HashSet<String>> dummyMap = createDummyMap();

        // Print the dummy map
        for (Map.Entry<String, HashSet<String>> entry : dummyMap.entrySet()) {
            String key = entry.getKey();
            HashSet<String> values = entry.getValue();
            System.out.println("Key: " + key + ", Values: " + values);
        }
    }
}
