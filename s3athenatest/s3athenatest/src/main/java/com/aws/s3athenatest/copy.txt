
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

public class AttributeConverter {

    public static void main(String[] args) {
        String inputFileName = "input.txt";
        String outputFileName = "output.txt";

        try {
            BufferedReader reader = new BufferedReader(new FileReader(inputFileName));
            FileWriter writer = new FileWriter(outputFileName);

            String className = reader.readLine().trim();
            String objectName = className.substring(0, 1).toLowerCase() + className.substring(1);

            writer.write(className + " " + objectName + " = new " + className + "();\n");

            String line;
            while ((line = reader.readLine()) != null) {
                String[] parts = line.trim().split("\\.");
                if (parts.length == 2) {
                    String attributeName = parts[1].replaceAll("_", "");
                    String methodName = "set" + attributeName.substring(0, 1).toUpperCase() + attributeName.substring(1);
                    writer.write(objectName + "." + methodName + "();\n");
                }
            }

            reader.close();
            writer.close();

            System.out.println("Conversion completed. Check output.txt");
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
