public static ResultSet convertCSVtoResultSet(String filePath) throws IOException, CsvException, SQLException {
        CSVReader reader = new CSVReader(new FileReader(filePath));
        List<String[]> rows = reader.readAll();
        reader.close();

        String[] header = rows.get(0);
        List<String[]> data = rows.subList(1, rows.size());

        // Create a custom ListResultSet
        ListResultSet resultSet = new ListResultSet(header);

        // Add rows to the result set
        for (String[] row : data) {
            resultSet.addRow(row);
        }

        // Return the ResultSet
        return resultSet;
    }
