
import java.io.*;
import java.util.zip.GZIPOutputStream;
import java.nio.charset.StandardCharsets;

public class GzipCompression {

    public static InputStream compressListToInputStream(List<String> inputList) throws IOException {
        ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();

        try (GZIPOutputStream gzipOutputStream = new GZIPOutputStream(byteArrayOutputStream)) {
            for (String item : inputList) {
                gzipOutputStream.write(item.getBytes(StandardCharsets.UTF_8));
                gzipOutputStream.write('\n'); // Add newline after each string for separation
            }
        }

        return new ByteArrayInputStream(byteArrayOutputStream.toByteArray());
    }

    // Example usage
    public static void main(String[] args) {
        List<String> inputList = Arrays.asList("String 1", "String 2", "String 3");

        try {
            InputStream compressedInputStream = compressListToInputStream(inputList);
            // Use the compressedInputStream as needed
            // For example, you can write it to a file or send it over a network.
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
