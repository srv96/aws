public static List<Map<String, String>> readExcelFile(String filePath) throws Exception {
        File file = new File(filePath);
        FileInputStream fis = new FileInputStream(file);
        
        DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
        DocumentBuilder db = dbf.newDocumentBuilder();
        
        Element root = db.parse(fis).getDocumentElement();
        NodeList sheetDataNodes = root.getElementsByTagName("sheetData");
        Element sheetData = (Element) sheetDataNodes.item(0);
        
        List<Map<String, String>> sheetDataList = new ArrayList<>();
        
        NodeList rowNodes = sheetData.getElementsByTagName("row");
        for(int i = 0; i < rowNodes.getLength(); i++) {
            Element row = (Element) rowNodes.item(i);
            NodeList cellNodes = row.getElementsByTagName("c");
            
            Map<String, String> rowMap = new HashMap<>();
            
            for(int j = 0; j < cellNodes.getLength(); j++) {
                Element cell = (Element) cellNodes.item(j);
                String columnName = getColumnFromCellReference(cell.getAttribute("r"));
                String cellValue = cell.getTextContent();
                
                rowMap.put(columnName, cellValue);
            }
            
            sheetDataList.add(rowMap);
        }
        
        fis.close();
        
        return sheetDataList;
    }
    
    private static String getColumnFromCellReference(String cellReference) {
        return cellReference.replaceAll("\\d", "");
    }
