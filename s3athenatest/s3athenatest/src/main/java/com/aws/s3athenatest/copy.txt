public static void writeLongListToFile(String fileName, List<Long> longList) {
        try (BufferedWriter bw = new BufferedWriter(new FileWriter(fileName))) {
            longList.forEach(number -> {
                try {
                    bw.write(number.toString() + "\n");
                } catch (IOException e) {
                    e.printStackTrace();
                }
            });
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
