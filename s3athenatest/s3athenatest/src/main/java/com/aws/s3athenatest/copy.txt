import java.io.File;
import java.util.ArrayList;
import java.util.List;

public class EmptyFileFinder {
    public static List<File> findEmptyFiles(String path) {
        List<File> emptyFiles = new ArrayList<>();
        
        File directory = new File(path);
        if (!directory.isDirectory()) {
            throw new IllegalArgumentException("Input path must be a directory.");
        }
        
        File[] files = directory.listFiles();
        if (files != null) {
            for (File file : files) {
                if (file.isFile() && file.length() == 0) {
                    emptyFiles.add(file);
                }
            }
        }
        
        return emptyFiles;
    }
    
    public static void main(String[] args) {
        String directoryPath = "path/to/your/directory";
        List<File> emptyFiles = findEmptyFiles(directoryPath);
        
        for (File file : emptyFiles) {
            System.out.println("Empty File: " + file.getAbsolutePath());
        }
    }
}
