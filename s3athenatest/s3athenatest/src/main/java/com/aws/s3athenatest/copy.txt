
import java.util.ArrayList;
import java.util.List;

public class CompactExclusiveRowFinder {

    public static List<List<String>> findExclusiveRows(List<List<String>> base, List<List<String>> target) {
        List<List<String>> result = new ArrayList<>();
        
        for (List<String> baseRow : base) {
            if (!target.contains(baseRow)) {
                baseRow.add("Base");
                result.add(baseRow);
            }
        }

        for (List<String> targetRow : target) {
            if (!base.contains(targetRow)) {
                targetRow.add("Target");
                result.add(targetRow);
            }
        }

        return result;
    }

    public static void main(String[] args) {
        List<List<String>> base = new ArrayList<>();
        List<List<String>> target = new ArrayList<>();

        // Populate base and target lists with data

        List<List<String>> exclusiveRows = findExclusiveRows(base, target);

        exclusiveRows.forEach(row -> System.out.println(row.get(0) + " - from " + row.get(1)));
    }
}
