import org.jsonschema2pojo.DefaultGenerationConfig;
import org.jsonschema2pojo.GenerationConfig;
import org.jsonschema2pojo.Jsonschema2Pojo;
import org.jsonschema2pojo.SchemaMapper;
import java.io.File;
import java.io.IOException;

public class JsonSchemaToPojoExample {
    public static void main(String[] args) {
        // Specify the input JSON schema file and the output directory for generated Java classes
        File inputJsonSchema = new File("schema.json");
        File outputDirectory = new File("generated");

        // Configure the generation options
        GenerationConfig config = new DefaultGenerationConfig();

        // Create a SchemaMapper instance
        SchemaMapper mapper = new SchemaMapper(config, new Jsonschema2Pojo());

        try {
            // Generate Java classes from the JSON schema
            mapper.generate(inputJsonSchema, outputDirectory);
            System.out.println("Java classes generated successfully.");
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
