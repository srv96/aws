public static ResultSet createResultSetFromFile(String filePath) throws SQLException {
        // Create an in-memory H2 database
        Connection connection = DriverManager.getConnection("jdbc:h2:mem:");

        // Load CSV data into the database
        Csv csv = new Csv();
        csv.setFieldSeparator(',');
        csv.setLineSeparator("\n");
        csv.setIgnoreCase(false);
        csv.setCharset("UTF-8");
        csv.setSchema("PUBLIC");
        csv.setTable("TEMP_TABLE");
        csv.setColumnNames(true);
        csv.setConnection(connection);
        csv.read(filePath, null);

        // Execute a select query to retrieve the data as a ResultSet
        Statement statement = connection.createStatement();
        ResultSet resultSet = statement.executeQuery("SELECT * FROM TEMP_TABLE");

        // Return the ResultSet
        return resultSet;
    }
