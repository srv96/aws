
import java.util.ArrayList;
import java.util.List;

public class FileNameProcessor {
    public static List<String> processFileNames(List<String> fileNames) {
        List<String> processedNames = new ArrayList<>();
        
        for (String fileName : fileNames) {
            String[] parts = fileName.split("_", 4);
            if (parts.length < 4) {
                throw new IllegalArgumentException("Invalid file name format: " + fileName);
            }
            
            String firstPart = parts[0] + "/" + parts[1] + "/" + parts[2] + "/" + parts[3].replace(".txt", "");
            processedNames.add(firstPart);
        }
        
        return processedNames;
    }
    
    public static void main(String[] args) {
        List<String> fileNames = new ArrayList<>();
        fileNames.add("tenants_ptr_omrap_hive_db_ptt_omrap_markets_confirmed_reg_exculsion_expr.txt");
        fileNames.add("tenants_ptr_omrap_hive_db_ptt_omrap_markets_semantic_derivative_profile_statement_contact.txt");
        
        List<String> processedNames = processFileNames(fileNames);
        
        for (String processedName : processedNames) {
            System.out.println(processedName);
        }
    }
}
