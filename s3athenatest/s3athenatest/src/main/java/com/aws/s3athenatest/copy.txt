import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.*;

public class CsvToGraph {
    public static Map<String, List<String>> readCsv(String fileName, String delimiter) throws IOException {
        Map<String, List<String>> graph = new HashMap<>();
        try (BufferedReader br = new BufferedReader(new FileReader(fileName))) {
            br.lines()
                    .map(line -> line.split(delimiter))
                    .filter(tokens -> tokens.length >= 2)
                    .forEach(tokens -> {
                        String parent = tokens[0].trim();
                        String child = tokens[1].trim();
                        graph.putIfAbsent(parent, new ArrayList<>());
                        graph.get(parent).add(child);
                    });
        }
        return graph;
    }

    public static void main(String[] args) {
        try {
            // Read the CSV file and create a graph
            Map<String, List<String>> graph = readCsv("input.csv", ",");
            System.out.println(graph);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
