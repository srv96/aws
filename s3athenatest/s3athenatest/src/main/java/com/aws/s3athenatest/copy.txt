public static void exportToExcel(List<Report> reports, String filePath) {
        try (Workbook workbook = new XSSFWorkbook()) {
            Sheet sheet = workbook.createSheet("Reports");

            // Writing the header
            Row headerRow = sheet.createRow(0);
            headerRow.createCell(0).setCellValue("Report ID");
            headerRow.createCell(1).setCellValue("Report Name");
            headerRow.createCell(2).setCellValue("SQL ID");
            headerRow.createCell(3).setCellValue("SQL Query");

            // Writing data
            int rowNum = 1;
            for (Report report : reports) {
                Row row = sheet.createRow(rowNum++);
                row.createCell(0).setCellValue(report.getReportId());
                row.createCell(1).setCellValue(report.getReportName());
                row.createCell(2).setCellValue(report.getSqlId());
                row.createCell(3).setCellValue(report.getSqlQuery());
            }

            // Write the workbook to the file
            try (FileOutputStream fileOut = new FileOutputStream(filePath)) {
                workbook.write(fileOut);
            }

        } catch (IOException e) {
            e.printStackTrace();
        }
    }
