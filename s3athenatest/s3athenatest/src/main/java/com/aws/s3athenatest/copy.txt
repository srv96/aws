public List<Map<String,String>> filterTable(List<Map<String,String>> table, List<String> headers) {
  List<Map<String,String>> filteredTable = new ArrayList<>();
  for (Map<String,String> row : table) {
    Map<String,String> filteredRow = new HashMap<>();
    for (String header : headers) {
      if (row.containsKey(header)) {
        filteredRow.put(header, row.get(header));
      }
    }
    filteredTable.add(filteredRow);
  }
  return filteredTable;
}
